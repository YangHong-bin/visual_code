function sayOrderPhone(){
  const answers = [];
  let list_card = [];
  
  selectOrder = getGlobal('selectOrder');
  
  if(selectOrder.restaurant_phone.startsWith('050')){
    answers.push(createTextFormat([
        '고객님,\n'
        +'주문하신 매장에서 직접 배달하기 때문에\n'
        +'해당 가게로 문의주셔야 빠르게 확인 가능합니다.\n'
        +'\n'
        +'가게 연락처 : '+ selectOrder.restaurant_phone+'\n'
        +'가게 전화가 불가능하실 경우 회원 주문 내역내 우측 하단 상담톡 버튼을 눌러주세요.\n'
    ]));
  }else{
    list_card.push(createCard(''
    ,'고객님,\n'
      +'가게 전화번호가 확인이 불가하여 아래 상담원 연결 버튼을 눌러주세요.\n'
      +'상세하고 빠르게 안내 도와드리겠습니다.'
    , ''
    ,[createTextButton('상담원 연결', '상담원 연결')]));
    
    answers.push(createCarousel(list_card));
  }
  answers.push(createButtonsFormat([createTextButton("처음으로", "처음으로")]));

  return answers;
}
